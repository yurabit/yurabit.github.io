下面贴上我后面自己重新写的代码吧（以后一定多多手敲代码，少用IDE）

import java.util.Scanner;

public class Main {
	public Node node;
	public static void main(String[] args) {
		int n;
		Main solution = new Main();
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		int[] arr = new int[n];
		for(int i = 0; i < n; i++) {
			arr[i] = sc.nextInt();
		}
		Node head = solution.sort(n, arr);
		head = head.next;
		while(head.next != null) {
			System.out.print(head.val + " ");
			head = head.next;
		}
		System.out.print(head.val);
	}

	public Node sort(int n, int[] arr) {
		for(int i = 0; i < n - 1; i++) {
			for(int j = 0; j < n - i - 1; j++) {
				if(arr[j] < arr[j + 1]) {
					swap(arr, j, j + 1);
				}
			}
		}
		Node head = new Node(0);
		for(int i = 0; i < n; i++) {
			Node temp = new Node(arr[i]);
			temp.next = head.next;
			head.next = temp;
		}
		return head;
	}

	private void swap(int[] arr, int a, int b) {
		int temp = arr[a];
		arr[a] = arr[b];
		arr[b] = temp;
	}
}

class Node {
	int val;
	Node next;
	Node(int val) {
		this.val = val;
	}
}
